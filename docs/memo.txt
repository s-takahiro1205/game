function renderLoop() {
    updateAnimations(); //なんだこれ
 
    if (dirty) {
        render();
        dirty = false;
    }
 
    requestAnimationFrame(renderLoop);
}
//ループでないなら playerとgameStateをproxy化 
let renderReserved = false;
 
function scheduleRender() {
    if (renderReserved) return;
 
    renderReserved = true;
 
    requestAnimationFrame(() => {
        renderReserved = false;
        render();
    });
}
 
const gameState = new Proxy(
    {},
    {
        set(target, key, value) {
            target[key] = value;
            scheduleRender(); //連続render対策
            return true;
        }
    }
);

数値計算（すべて切り捨て）
攻撃
ダイス数：速度 / 333 + 装備の値 + 1：下限1 999で3
ダイス面：攻撃の25% + 装備の値：下限1 999で250
修正値：器用 ^ 0.80 - 1 + 装備の値：下限0 999で250
素手だと3D250+250で最大で1000出せる
 
防御
防御値：(防御 + 体格/10) / 5 + 装備の値：下限0
回避値：速さ / 20 + (1001 - 体格 ^ 1.5) / 100 + 装備の値：下限0上限99
 
ダメージ計算
ダメージ：攻撃のダイスロール - 防御値：下限0
回避判定：1D100 < 防御側回避値 - 攻撃側回避値
