function getStatus(chara, property) {
  const base = chara[property];
  if (!chara.statusEffects || chara.statusEffects.length === 0) return base;

  let multiplier = 1.0;
  let flat = 0;

  for (const effect of chara.statusEffects) {
    const mod = DEBUFF_STATUS_MODIFIERS[effect.type]?.[property];
    if (!mod) continue;
    if (mod.rate)  multiplier += mod.rate; // 例: -0.2で20%減
    if (mod.flat)  flat += mod.flat;
  }

  return Math.max(0, Math.floor(base * multiplier) + flat);
}

例えば、
主人公にのみ「捕縛スキル」を追加して、体力が残り20％(切り捨て,下限1)を切った相手に使用可能。
命中したら相手を「捕縛：1」の状態異常にする。捕縛状態はターン経過で減少しない。
「捕縛」使用後に3以上蓄積したら捕縛成功で戦闘からそのキャラを離脱させる。
捕縛状態のキャラは捕縛状態を1減らす「もがく」コマンドが利用可能になる。
（状態異常にターン減衰なしが必要）
 
こういった形なら、
主人公は「捕縛」を使い続けないといけない（ハンデを背負う）、
捕縛状態のキャラは「もがく」かHPを回復することで逃れられる、
「もがく」の無効化のためには睡眠や麻痺などの状態異常にして弱らせるのが有効である、
捕縛したい敵がいるならその敵以外はあらかじめ倒す、
弱い敵を捕縛したい場合は全体攻撃を控える、
といった形で戦略性が生まれそう。

数値計算（すべて切り捨て）
攻撃
ダイス数：速度 / 333 + 装備の値 + 1：下限1 999で3
ダイス面：攻撃の25% + 装備の値：下限1 999で250
修正値：器用 ^ 0.80 - 1 + 装備の値：下限0 999で250
素手だと3D250+250で最大で1000出せる
 
防御
防御値：(防御 + 体格/10) / 5 + 装備の値：下限0
回避値：速さ / 20 + (1001 - 体格 ^ 1.5) / 100 + 装備の値：下限0上限99
 
ダメージ計算
ダメージ：攻撃のダイスロール - 防御値：下限0
回避判定：1D100 < 防御側回避値 - 攻撃側回避値
